def kebabcase
  self.gsub(/([A-Z]+)([A-Z][a-z])/,'\1-\2')
      .gsub(/([a-z\d])([A-Z])/,'\1-\2')
      .gsub(/[\s_\/\-_\.]+/, '-')
      .downcase
end
